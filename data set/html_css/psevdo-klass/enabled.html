Псевдокласс :enabled

Псевдокласс :enabled используется для применения стиля к доступным (не заблокированным) элементам форм. По умолчанию, все элементы форм являются доступными, если в коде HTML к ним не добавляется атрибут disabled.

.elem:enabled { ... }
Типовой договор
<script>
  function agreeForm(f) {
   if (f.agree.checked) f.submit.disabled = 0;  // submit - name="submit"
   else f.submit.disabled = 1;                  // берет из псевдомассива form
  }
</script>
<form action="./">
  <p>Типовой договор</p>
  <input type="checkbox" id="agree" onclick="agreeForm(this.form)">
  <label for="agree">Я согласен со всеми условиями</label>
  <p><input type="submit" name="submit" value="Отправить" disabled></p>
</form>

input:enabled { background: rgb(72, 252, 183); }
