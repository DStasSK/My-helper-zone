Псевдокласс :optional

Псевдокласс :optional применяет стилевые правила к элементу input, у которого не задан атрибут required (он позволяет выделять поля обязательные к заполнению перед отправкой формы). Таким образом :optional применяется к необязательным полям формы.

input:optional {  }

input:required {border:2px solid red}
input:optional {background-color: greenyellow}

<form action="handler.php" id="optional">
  <p>
    <label for="user">Имя</label>
    <input id="user" type="text" required>
  </p>
  <p>
    <label for="email">E-mail</label>
    <input id="email" type="text" optional>
  </p>
  <p>
    <label for="">Разное</label>
    <input type="checkbox">
    <input type="radio" name="radio_1">
    <input type="radio" name="radio_1">
    <input type="radio" name="radio_1">
  </p>
  <p>
    <label for="text">Комментарий</label>
    <textarea cols="40" rows="8" id="text"></textarea>
  </p>
  <p>
    <label> </label>
    <input type="submit" id="send" value="Отправить">
  </p>
</form>
