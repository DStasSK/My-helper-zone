пример 1 - сужение

Блок <div> будет изначально с шириной 700px и за 5 секунд уменьшится до 100px.

сужение
HTML 
<div class="pr1"><span>с</span>...<span>е</span></div>

CSS 
.pr1{margin:0px auto; background:red; height:50px; width:700px;

  animation-name:animI; /* название ключевых кадров (см.ниже) */
  animation-duration:5s;            /* длительность анимации */
  animation-iteration-count:infinite;
}

.pr1 span {
  width:calc(100% / 7); text-align:center;
  display:inline-block; line-height:50px;
}

@keyframes animI {
  from {width:700px;} /* первый ключевой кадр, ширина блока 700px */
  to {width:100px;}}  /* последний ключевой кадр, ширина - 100px */

.pr4 p:nth-child(1) span{animation:IVa 1.05s linear infinite;}
/* 158рх/150рх = 1.05 секунд * 12 / 8 = 1.57сек */
.pr4 p:nth-child(2) span{animation:IVb 1.57s linear infinite;}

/* отключаем анимацию при наведении */
.pr4 p:hover {animation-play-state: paused;}
.pr4 p span:hover {animation-play-state: paused;}

@keyframes IVa {
    0% {transform:rotateZ(0deg);}
  100% {transform:rotateZ(-360deg);}}

@keyframes IVb {
    0% {transform:rotateZ(0deg);}
  100% {transform:rotateZ(360deg);}}
